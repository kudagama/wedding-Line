import os
from PIL import Image

# Input and output folders inside Google Drive
input_folder = '/content/drive/My Drive/ImagesToGo'
output_folder = '/content/drive/My Drive/OptimizedImages'

# Make sure output folder exists
os.makedirs(output_folder, exist_ok=True)

# Loop through files and compress
for filename in os.listdir(input_folder):
    if filename.lower().endswith(('.jpg', '.jpeg', '.png')):
        img_path = os.path.join(input_folder, filename)
        img = Image.open(img_path)

        # Save compressed version (JPEG quality ~70)
        output_path = os.path.join(output_folder, filename)
        img.save(output_path, optimize=True, quality=70)

        print(f"Compressed: {filename} â†’ {output_path}")
